function getAirResistanceForce(): number {
  const c = 0.1 // constant c
  const objectSpeed = this.velocity.magnitude() // v
  const dragMagnitude = c * objectSpeed * objectSpeed // 공식의 첫 부분
  const oppositeDirection = this.velocity.direction().inverse() // 공식의 둘째 부분
  return dragMagnitude * oppositeDirection
}
